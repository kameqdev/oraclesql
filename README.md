DECLARE
    V_EMP_ID EMPLOYEES.EMPLOYEE_ID%TYPE := &PODAJ_ID_PRACOWNIKA; 
    V_DEPT_NAME DEPARTMENTS.DEPARTMENT_NAME%TYPE;
    V_DEPT_ID EMPLOYEES.DEPARTMENT_ID%TYPE;
    V_EMP_COUNT NUMBER;
BEGIN
    SELECT d.DEPARTMENT_NAME, e.DEPARTMENT_ID 
    INTO V_DEPT_NAME, V_DEPT_ID
    FROM EMPLOYEES e
    JOIN DEPARTMENTS d ON e.DEPARTMENT_ID = d.DEPARTMENT_ID
    WHERE e.EMPLOYEE_ID = V_EMP_ID;
    
    DBMS_OUTPUT.PUT_LINE('Pracownik należy do działu: ' || V_DEPT_NAME);

    SELECT COUNT(*) 
    INTO V_EMP_COUNT
    FROM EMPLOYEES
    WHERE DEPARTMENT_ID = V_DEPT_ID;
    
    IF V_EMP_COUNT > 10 THEN
        DBMS_OUTPUT.PUT_LINE('Dział zatrudnia więcej niż 10 pracowników (dokładnie: ' || V_EMP_COUNT || ').');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Dział NIE zatrudnia więcej niż 10 pracowników (dokładnie: ' || V_EMP_COUNT || ').');
    END IF;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Brak danych - nie ma pracownika o ID: ' || V_EMP_ID || ' lub nie jest on przypisany do żadnego działu.');
END;
/

DECLARE
    V_EMP_ID EMPLOYEES.EMPLOYEE_ID%TYPE := &PODAJ_ID_PRACOWNIKA; 
    V_JOB_TITLE JOBS.JOB_TITLE%TYPE;
    V_JOB_ID EMPLOYEES.JOB_ID%TYPE;
    V_SALARY EMPLOYEES.SALARY%TYPE;
    V_AVG_SALARY NUMBER;
BEGIN
    SELECT j.JOB_TITLE, e.JOB_ID, e.SALARY
    INTO V_JOB_TITLE, V_JOB_ID, V_SALARY
    FROM EMPLOYEES e
    JOIN JOBS j ON e.JOB_ID = j.JOB_ID
    WHERE e.EMPLOYEE_ID = V_EMP_ID;
    
    DBMS_OUTPUT.PUT_LINE('Pracownik zajmuje stanowisko: ' || V_JOB_TITLE);

    SELECT AVG(SALARY) 
    INTO V_AVG_SALARY
    FROM EMPLOYEES
    WHERE JOB_ID = V_JOB_ID;
    
    IF V_SALARY > V_AVG_SALARY THEN
        DBMS_OUTPUT.PUT_LINE('Pracownik zarabia powyżej średniej dla tego stanowiska (Pensja: ' || V_SALARY || ', Średnia: ' || ROUND(V_AVG_SALARY, 2) || ').');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Pracownik NIE zarabia powyżej średniej dla tego stanowiska (Pensja: ' || V_SALARY || ', Średnia: ' || ROUND(V_AVG_SALARY, 2) || ').');
    END IF;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Błąd - nie odnaleziono pracownika o identyfikatorze: ' || V_EMP_ID);
END;
/

DECLARE

    V_JOB_TITLE JOBS.JOB_TITLE%TYPE;
    V_JOB_ID EMPLOYEES.JOB_ID%TYPE;
    V_COUNT NUMBER;
BEGIN
    SELECT j.JOB_TITLE, e.JOB_ID 
    INTO V_JOB_TITLE, V_JOB_ID
    FROM EMPLOYEES e
    JOIN JOBS j ON e.JOB_ID = j.JOB_ID
    WHERE e.EMPLOYEE_ID = V_EMP_ID;

    INTO V_COUNT

    WHERE JOB_ID = V_JOB_ID;

        DBMS_OUTPUT.PUT_LINE('Pracuje tutaj więcej niż 5 osób (dokładnie: ' || V_COUNT || ').');
    ELSE
        DBMS_OUTPUT.PUT_LINE('NIE pracuje tutaj więcej niż 5 osób (dokładnie: ' || V_COUNT || ').');
    END IF;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Pracownik o ID ' || V_EMP_ID || ' nie istnieje w bazie.');
END;
